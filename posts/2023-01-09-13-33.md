## Mapping Modifiersを使う

```ts
type NotOptionalType<T> = {
  [P in keyof T]-?: T[P];
};

type WritableType<T> = {
  -readonly [P in keyof T]: T[P];
};
```

## 一部のプロパティだけ差し替える

```ts
type WritablePartType<T, K extends keyof T> = {
  -readonly [P in K]: T[P];
} & Omit<T, K>;
```

## キー名を差し替える

```ts
type Getters<T> = {
  [K in keyof T as `get${Capitalize<string & K>}`]: () => T[K]
};

type T2 = Getters<T1>;
```

## 値からキーを作る

```ts
type Foo = { id: "foo" };
type Bar = { id: "bar" };

type T4<T extends { id: string }> = {
  [P in T as T['id']]: void;
};

type T5 = T4<Foo>;
```
