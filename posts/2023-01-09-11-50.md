## オブジェクトの一部プロパティだけモックする

```ts
type RequestBody = {
  a: string;
  b: string;
  c: number;
  d: boolean;
};

it('', () => {
// Type Assertionを使ったありがちな方法
// 型定義が変わった時に追随できない
//   const body1: RequestBody = {
//     a: '',
//   } as unknown as RequestBody;
// });

it('', () => {
  const body2: Pick<RequestBody, 'a' | 'b'> = {
    a: '',
    b: '',
  };
});
```
