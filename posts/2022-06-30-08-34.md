```ts
const CACHE_TIMEOUT = 2000;
let users$: Observable<string[]>;

// old
function getUsers() {
  if (users$) {
    return users$;
  } else {
    users$ = ajax.getJSON();
    setTimeout(() => {
      users$ = undefined;
    }, CACHE_TIMEOUT);
    return users$;
  }
}

// new
const users$ = ajax.getJSON()
  .pipe(
    share({
      connector: () => new ReplySubject(1),
      resetOrComplete: () => timer(CACHE_TIMEOUT),
    })
  )
;
```
