```ts
type T1 = {
  a: string;
  b: string;
  c: string;
};

type T2 = Extract<keyof T1, 'a'>;
// 'a'

type T3 = Exclude<keyof T1, 'a'>;
// 'b' | 'c'

type T4 = Pick<T1, 'a'>;
// { a: string }

type T5 = Omit<T1, 'a'>;
// { b: string; c: string }
```
