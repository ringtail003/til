## Promiseを返す関数の戻り値をUnionで宣言する

```ts
declare function fn1 (): Promise<string>;
declare function fn2 (): Promise<number>;

const V1: Awaited<ReturnType<typeof fn1> | ReturnType<typeof fn2>> = await fn1();
// string | number
```
