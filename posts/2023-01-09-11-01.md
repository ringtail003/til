## functionの戻り値をletに格納して取り回すケース

```ts
const fn = (): [string, number] => {
  return ['a', 1];
};

type Result = ReturnType<typeof fn>;

describe('', () => {
  let result: Result;

  it('', () => {
    result = fn();
  });
});
```
