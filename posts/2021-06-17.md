```ts
  beforeEach(() => {
    jest.spyOn(repository, "fetch").mockReturnValue(Rx.of().pipe(delay(1)));
  });

  it("50件ごとにリクエストする事", fakeAsync(() => {
    let count = 0;

    sut.valueChanges$.subscribe(() => {
      count++;
    });

    sut.fetch().subscribe();
    tick(1);
    expect(count).toBe(1);

    sut.fetch().subscribe();
    tick(1);
    expect(count).toBe(2);

    sut.fetch().subscribe();
    tick(1);
    expect(count).toBe(3);

    // 不要なタスクを削除する
    discardPeriodicTasks();
  }));
});
```
