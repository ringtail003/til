## Brandedでないクラス

オブジェクトの構造が一致すれば同じ型とみなされてしまう。

```ts
class Foo {
  constructor(id: string) {}
}

class Bar {
  constructor(id: string) {}
}

declare function acceptFoo(prop: Foo): void;

const foo = new Foo('');
const bar = new Bar('');

acceptFoo(foo); // OK
acceptFoo(bar); // OK
```

## Brandedなクラス

```ts
class Foo {
  _fooBland!: never;
  constructor(id: string) {}
}

class Bar {
  _barBland!: never;
  constructor(id: string) {}
}

declare function acceptFoo(prop: Foo): void;

const foo = new Foo('');
const bar = new Bar('');

acceptFoo(foo); // OK
acceptFoo(bar); // Error
```

## Brandedなオブジェクト

```ts
type Brand<K, T> = K & { __brand: T }

type Foo = {
  id: string;
};
type Bar = {
  id: string;
};

type BrandedFoo = Brand<Foo, 'Foo'>;
type BrandedBar = Brand<Bar, 'Bar'>;

declare const foo: BrandedFoo;
declare const bar: BrandedBar;

declare function acceptFoo(prop: BrandedFoo): void;

acceptFoo(foo); // OK
acceptFoo(bar); // Error
```
