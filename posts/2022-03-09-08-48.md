```bash
# PEMフォーマットのキーを作る
ssh-keygen -m PEM
> Enter file which…: hoge.pem
> Enter passphrase: {passphrase}
```

```bash
# コピー
pbcopy < hoge.pem.pub

# GitHubに登録する時
open https://github.com/settings/keys

# 公開鍵をサーバーのSSH認証に使う時
scp -i ./hoge ./hoge.pem.pub {user}@{ip}:~/.ssh

ssh {user}@{ip} -I ~/.ssh/{接続可能な秘密鍵}
cat hoge.pem.pub >> ~/.ssh/authrized_key
```
